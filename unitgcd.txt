#include <bits/stdc++.h>

using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    int t;
    cin>>t;
    while(t--)
    {
       int n;
       cin>>n;
       if(n==1)
       {
           cout<<1<<"\n";
           cout<<1<<" "<<1<<"\n";
       }
       else
       {
       if(n%2==0)
       {
            int a[n],b[n];
           cout<<n/2<<"\n";
           int j=0,k=0;
             for(int i=1;i<=n;i++)
           {
               if(i%2==0)
               {
                   a[j]=i;
                   j++;
               }
               else
               {
                   b[k]=i;
                   k++;
               }
           }
           for(int i=0;i<j;i++)
           {
               cout<<2<<" "<<a[i]<<" "<<b[i]<<"\n";
           }
           
       }
       else
       {
           cout<<n/2<<"\n";
           int a[n],b[n];
           int j=0,k=0;
           for(int i=1;i<=n-1;i++)
           {
               if(i%2==0)
               {
                   a[j]=i;
                   j++;
               }
               else
               {
                   b[k]=i;
                   k++;
               }
           }
           cout<<3<<" "<<1<<" "<<2<<" "<<n<<"\n";
           for(int i=1;i<j;i++)
           {
               cout<<2<<" "<<a[i]<<" "<<b[i]<<"\n";
           }
       
       }
       }
    }

    return 0;
}
