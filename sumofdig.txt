#include <bits/stdc++.h>
using namespace std;
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    string str;
    cin>>str;
    long long int n;
    n=str.length();
    int sum=0;
    if(n==1)
    cout<<0<<endl;
    else
    {
  
    for(int i=0;i<n;i++)
    {
        sum=sum+(str[i]-48);
    }
      int s=sum;
      int c=1;
      while(s>=10)
      {
          s=0;
         while(sum>0)
        {
          s=s+sum%10;
          sum=sum/10;
        }
        
         sum=s;
          c++;
      }
      
      cout<<c<<endl;
      
     }
    
    return 0;
}
